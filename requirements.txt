        <div class="bg-white/10 rounded-xl p-3 max-w-lg">
          <p class="text-sm"><span class="font-semibold text-amber-400">Shubh AI:</span> Welcome. Please upload your palm image and ask your questions.</p>

      <!-- Input -->
      <div class="mt-4 flex gap-2">
        <input id="chatInput" type="text" placeholder="Ask about marriage, career, health..." class="flex-1 rounded-xl bg-black/40 border border-white/10 px-4 py-3 focus:outline-none focus:ring-2 focus:ring-amber-400" />

  <!-- Scripts -->
  <script>
    const palmInput = document.getElementById('palmInput');
    const previewBox = document.getElementById('previewBox');
    const previewImage = document.getElementById('previewImage');
    const chatBox = document.getElementById('chatBox');
    const chatInput = document.getElementById('chatInput');

    palmInput.addEventListener('change', () => {
      const file = palmInput.files[0];
      if (!file) return;
      previewImage.src = URL.createObjectURL(file);
      previewBox.classList.remove('hidden');

      addMessage('Shubh AI', 'Palm image received. You may now ask your questions.');
    });

    function sendMessage() {
      const text = chatInput.value.trim();
      if (!text) return;

      addMessage('You', text, true);
      chatInput.value = '';

      setTimeout(() => {
        addMessage('Shubh AI', 'Based on your palm, this question relates to personal tendencies. A detailed response will appear here after analysis.');
      }, 800);
    }

    function addMessage(sender, text, isUser = false) {
      const div = document.createElement('div');
      div.className = `rounded-xl p-3 max-w-lg ${isUser ? 'bg-amber-400 text-black ml-auto' : 'bg-white/10'}`;
      div.innerHTML = `<p class="text-sm"><span class="font-semibold ${isUser ? '' : 'text-amber-400'}">${sender}:</span> ${text}</p>`;
      chatBox.appendChild(div);
      chatBox.scrollTop = chatBox.scrollHeight;
    }
  </script>
